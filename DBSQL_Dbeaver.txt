-- 1. Hapus tabel barang lama agar strukturnya bersih kembali
DROP TABLE IF EXISTS barang CASCADE;

-- 2. Buat Ulang Tabel Barang dengan Klasifikasi Super Lengkap
CREATE TABLE barang (
    id SERIAL PRIMARY KEY,
    kode_barang VARCHAR(20) UNIQUE NOT NULL,
    nama_barang VARCHAR(150) NOT NULL,          -- Nama spesifik unit (Contoh: MacBook Pro M3, iPhone 15)
    jenis_komoditas VARCHAR(100) NOT NULL,      -- Sub-Kategori (Contoh: Ponsel & Komunikasi, Peralatan Dapur)
    kategori VARCHAR(100) NOT NULL,             -- Kategori Utama (Contoh: 1. Consumer Electronics)
    merk VARCHAR(50) NOT NULL,                  -- Brand (Contoh: Google, Iphone, Xiaomi, Samsung, Asus, dll.)
    stok INT NOT NULL DEFAULT 0,
    harga NUMERIC(15, 2) NOT NULL,
    foto_barang VARCHAR(255) DEFAULT 'default.jpg',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
